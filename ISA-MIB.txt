isaMIBObjects     OBJECT IDENTIFIER ::= { isaMIB 1 }
isaMIBConformance OBJECT IDENTIFIER ::= { isaMIB 2 }


-- define 3 objects, which will all be implemented in different ways
-- within the tutorial.

isaAgentModules   OBJECT IDENTIFIER ::= { isaMIBObjects 1 }


loginObject OBJECT-TYPE
    SYNTAX      DisplayString (SIZE (0..255))
    MAX-ACCESS  read-only
    STATUS      current
    DESCRIPTION
	"String that represents login"
    ::= { isaAgentModules 1 }

currentTimeObject OBJECT-TYPE
    SYNTAX      DisplayString (SIZE (0..255))
    MAX-ACCESS  read-only
    STATUS      current
    DESCRIPTION
	"String representing current time"
    ::= { isaAgentModules 2 }

numberObject OBJECT-TYPE
    SYNTAX      Integer32
    MAX-ACCESS  read-write
    STATUS      current
    DESCRIPTION
	"Represents a read/write number"
    ::= { isaAgentModules 3 }

operatingSystemObject OBJECT-TYPE
    SYNTAX      DisplayString (SIZE (0..255))
    MAX-ACCESS  read-only
    STATUS      current
    DESCRIPTION
	"String representing information about operating system"
    ::= { isaAgentModules 4 }
